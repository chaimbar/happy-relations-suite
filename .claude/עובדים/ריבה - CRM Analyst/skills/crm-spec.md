### פלט ב: supabase-schema.sql
```sql
-- ================================================
-- CRM Schema — [project-name]
-- הרץ ב-Supabase SQL Editor
-- ================================================

-- Extensions
create extension if not exists "uuid-ossp";

-- Enums
create type [entity]_status as enum ('[val1]', '[val2]', '[val3]');

-- Tables
create table if not exists [entity] (
  id uuid default gen_random_uuid() primary key,
  created_at timestamptz default now() not null,
  updated_at timestamptz default now() not null,
  -- שדות ספציפיים
  [field] [type] [constraints],
  -- foreign keys
  [relation_id] uuid references [other_table](id) on delete cascade,
  -- owner
  user_id uuid references auth.users(id) on delete cascade not null
);

-- Indexes
create index if not exists idx_[entity]_user_id on [entity](user_id);
create index if not exists idx_[entity]_status on [entity](status);
create index if not exists idx_[entity]_created_at on [entity](created_at desc);

-- RLS
alter table [entity] enable row level security;

-- Policies
create policy "[entity]_select_own" on [entity]
  for select using (auth.uid() = user_id);
create policy "[entity]_insert_own" on [entity]
  for insert with check (auth.uid() = user_id);
create policy "[entity]_update_own" on [entity]
  for update using (auth.uid() = user_id);
create policy "[entity]_delete_own" on [entity]
  for delete using (auth.uid() = user_id);

-- Updated_at trigger
create or replace function handle_updated_at()
returns trigger language plpgsql as $$
begin
  new.updated_at = now();
  return new;
end;
$$;

create trigger [entity]_updated_at
  before update on [entity]
  for each row execute function handle_updated_at();

```
